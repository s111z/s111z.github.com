滑雪

这道题对于每个位置只能滑向更低的位置。

状态表示：用f[i][j]来表示从（i，j）位置出发的所有路径，值表示路径和的最大值。

状态计算：通过四个方向的最大值+1计算得到。

代码：

#include<bits/stdc++.h>
using namespace std;
int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};
int f[400][400],a[400][400];

int n,m;
int dp(int x,int y)
{
    if(f[x][y]!=-1) return f[x][y];
    
    f[x][y]=1;
    
    for(int i=0;i<4;i++)
    {
        int nx=x+dx[i],ny=y+dy[i];
        if(1<=nx&&nx<=n&&1<=ny&&ny<=m&&a[nx][ny]<a[x][y])
        {
            f[x][y]=max(f[x][y],dp(nx,ny)+1);
        }
    }
    return f[x][y];
}
int main()
{
    
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            scanf("%d",&a[i][j]);
            
    memset(f,-1,sizeof f);
    int res=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            res=max(dp(i,j),res);
        }
    }
    printf("%d",res);
}
